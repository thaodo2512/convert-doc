# 4. STRINGS (Prioritize explicit encoding key)
elif key_schema.get('type') == 'string' or 'string' in desc:
    # A. Check explicit 'x-binary-encoding' first (Most Reliable)
    encoding = key_schema.get('x-binary-encoding', '').lower()
    
    if encoding == 'utf-16be':
        field_type = 'strUTF-16'  # Or 'strUTF-16BE' if you prefer
    elif encoding == 'utf-16le':
        field_type = 'strUTF-16LE'
    elif encoding == 'us-ascii':
        field_type = 'ascii'
    elif encoding == 'utf-8':
        field_type = 'utf-8'
    
    # B. Fallback: Parse description (Backward Compatibility)
    elif 'ascii' in desc:
        field_type = 'ascii'
    elif 'utf-16' in desc or 'strutf-16' in desc:
        field_type = 'strUTF-16'
    elif 'unicode be16' in desc:
        field_type = 'strunicode be16'
    elif 'unicode le16' in desc:
        field_type = 'strunicode le16'
    elif 'utf-8' in desc:
        field_type = 'utf-8'
    else:
        field_type = 'strASCII'  # Final Default
